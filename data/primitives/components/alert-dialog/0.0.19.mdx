```jsx hero
import React from 'react';
import { styled, keyframes } from '@stitches/react';
import { violet, red, mauve } from '@radix-ui/colors';
import * as AlertDialogPrimitive from '@radix-ui/react-alert-dialog';

const fadeIn = keyframes({
  from: { opacity: 0 },
  to: { opacity: 1 },
});

const fadeOut = keyframes({
  from: { opacity: 1 },
  to: { opacity: 0 },
});

const StyledOverlay = styled(AlertDialogPrimitive.Overlay, {
  backgroundColor: 'hsla(0 0% 0% / 10%)',
  position: 'fixed',
  inset: 0,
  '&[data-state=open]': { animation: `${fadeIn} 300ms ease-out` },
  '&[data-state=closed]': { animation: `${fadeOut} 300ms ease-out` },
});

function Root({ children, ...props }) {
  return (
    <AlertDialogPrimitive.Root {...props}>
      <StyledOverlay />
      {children}
    </AlertDialogPrimitive.Root>
  );
}

const StyledContent = styled(AlertDialogPrimitive.Content, {
  backgroundColor: 'white',
  borderRadius: '8px',
  boxShadow:
    'hsl(206 22% 7% / 35%) 0px 10px 38px -10px, hsl(206 22% 7% / 20%) 0px 10px 20px -15px',
  position: 'fixed',
  top: '50%',
  left: '50%',
  transform: 'translate(-50%, -50%)',
  width: '90vw',
  maxWidth: '500px',
  maxHeight: '85vh',
  padding: '20px',
  marginTop: '-5vh',
  '&[data-state=open]': { animation: `${fadeIn} 200ms ease-out` },
  '&[data-state=closed]': { animation: `${fadeOut} 100ms ease-out` },

  '&:focus': { outline: 'none' },
});

const StyledTitle = styled(AlertDialogPrimitive.Title, {
  margin: 0,
  fontWeight: 500,
  color: mauve.mauve12,
  fontSize: '17px',
});

const StyledDescription = styled(AlertDialogPrimitive.Description, {
  margin: '10px 0 20px',
  color: mauve.mauve11,
  fontSize: '15px',
});

// Exports
const AlertDialog = Root;
const AlertDialogTrigger = AlertDialogPrimitive.Trigger;
const AlertDialogContent = StyledContent;
const AlertDialogTitle = StyledTitle;
const AlertDialogDescription = StyledDescription;
const AlertDialogAction = AlertDialogPrimitive.Action;
const AlertDialogCancel = AlertDialogPrimitive.Cancel;

// On your app...
const Button = styled('button', {
  all: 'unset',
  borderRadius: '10px',
  padding: '10px 15px',
  height: '20px',
  display: 'inline-flex',
  alignItems: 'center',
  justifyContent: 'center',
  fontSize: '15px',

  variants: {
    variant: {
      violet: {
        backgroundColor: violet.violet3,
        color: violet.violet10,
        '&:hover': { backgroundColor: violet.violetA9 },
        '&:focus': { boxShadow: `0 0 0 2px ${violet.violet8}` },
      },
      red: {
        backgroundColor: red.red3,
        color: red.red10,
        '&:hover': { backgroundColor: red.red4 },
        '&:focus': { boxShadow: `0 0 0 2px ${red.red8}` },
      },
      ghost: {
        backgroundColor: 'transparent',
        color: mauve.mauve10,
        '&:hover': { backgroundColor: mauve.mauve4 },
        '&:focus': { boxShadow: `0 0 0 2px ${mauve.mauve8}` },
      },
    },
  },

  defaultVariants: {
    variant: 'violet',
  },
});

export default () => (
  <AlertDialog>
    <AlertDialogTrigger as={Button}>
      Delete account
    </AlertDialogTrigger>
    <AlertDialogContent>
      <AlertDialogTitle>Are you absolutely sure?</AlertDialogTitle>
      <AlertDialogDescription>
        This action cannot be undone. This will permanently delete
        your account and remove your data from our servers.
      </AlertDialogDescription>
      <AlertDialogCancel
        as={Button}
        variant="ghost"
        css={{ marginRight: '10px' }}
      >
        Cancel
      </AlertDialogCancel>
      <AlertDialogAction as={Button} variant="red">
        Yes, delete account
      </AlertDialogAction>
    </AlertDialogContent>
  </AlertDialog>
);
```

## Installation

Install the component from your command line.

```bash
npm install @radix-ui/react-alert-dialog
```

## Anatomy

Import the components and piece the parts together.

```jsx
import * as AlertDialog from '@radix-ui/react-alert-dialog';

export default () => (
  <AlertDialog.Root>
    <AlertDialog.Trigger />
    <AlertDialog.Overlay />
    <AlertDialog.Content>
      <AlertDialog.Title />
      <AlertDialog.Description />
      <AlertDialog.Cancel />
      <AlertDialog.Action />
    </AlertDialog.Content>
  </AlertDialog.Root>
);
```
